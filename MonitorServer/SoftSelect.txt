SELECT 
    no,
    logtime,
    CASE serverno
        WHEN 0 THEN 'CHAT'
        WHEN 1 THEN 'LOGIN'
        WHEN 2 THEN 'GAME'
        WHEN 3 THEN 'HARDWARE'
    END AS ServerNo,
    CASE type
		WHEN 1 THEN 'dfMONITOR_DATA_TYPE_LOGIN_SERVER_RUN'
        WHEN 2 THEN 'dfMONITOR_DATA_TYPE_LOGIN_SERVER_CPU'
        WHEN 3 THEN 'dfMONITOR_DATA_TYPE_LOGIN_SERVER_MEM'
        WHEN 4 THEN 'dfMONITOR_DATA_TYPE_LOGIN_SESSION'
        WHEN 5 THEN 'dfMONITOR_DATA_TYPE_LOGIN_AUTH_TPS'
        WHEN 6 THEN 'dfMONITOR_DATA_TYPE_LOGIN_PACKET_POOL'
        WHEN 10 THEN 'dfMONITOR_DATA_TYPE_GAME_SERVER_RUN'
        WHEN 11 THEN 'dfMONITOR_DATA_TYPE_GAME_SERVER_CPU'
        WHEN 12 THEN 'dfMONITOR_DATA_TYPE_GAME_SERVER_MEM'
        WHEN 13 THEN 'dfMONITOR_DATA_TYPE_GAME_SESSION'
        WHEN 14 THEN 'dfMONITOR_DATA_TYPE_GAME_AUTH_PLAYER'
        WHEN 15 THEN 'dfMONITOR_DATA_TYPE_GAME_GAME_PLAYER'
		WHEN 16 THEN 'dfMONITOR_DATA_TYPE_GAME_ACCEPT_TPS'
        WHEN 17 THEN 'dfMONITOR_DATA_TYPE_GAME_PACKET_RECV_TPS'
        WHEN 18 THEN 'dfMONITOR_DATA_TYPE_GAME_PACKET_SEND_TPS'
        WHEN 19 THEN 'dfMONITOR_DATA_TYPE_GAME_DB_WRITE_TPS'
        WHEN 20 THEN 'dfMONITOR_DATA_TYPE_GAME_DB_WRITE_MSG'
        WHEN 21 THEN 'dfMONITOR_DATA_TYPE_GAME_AUTH_THREAD_FPS'
        WHEN 22 THEN 'dfMONITOR_DATA_TYPE_GAME_GAME_THREAD_FPS'
        WHEN 23 THEN 'dfMONITOR_DATA_TYPE_GAME_PACKET_POOL'
        WHEN 30 THEN 'dfMONITOR_DATA_TYPE_CHAT_SERVER_RUN'
        WHEN 31 THEN 'dfMONITOR_DATA_TYPE_CHAT_SERVER_CPU'
        WHEN 32 THEN 'dfMONITOR_DATA_TYPE_CHAT_SERVER_MEM'
        WHEN 33 THEN 'dfMONITOR_DATA_TYPE_CHAT_SESSION'
        WHEN 34 THEN 'dfMONITOR_DATA_TYPE_CHAT_PLAYER'
        WHEN 35 THEN 'dfMONITOR_DATA_TYPE_CHAT_UPDATE_TPS'
        WHEN 36 THEN 'dfMONITOR_DATA_TYPE_CHAT_PACKET_POOL'
        WHEN 37 THEN 'dfMONITOR_DATA_TYPE_CHAT_UPDATEMSG_POOL'
        WHEN 40 THEN 'dfMONITOR_DATA_TYPE_MONITOR_CPU_TOTAL'
        WHEN 41 THEN 'dfMONITOR_DATA_TYPE_MONITOR_NONPAGED_MEMORY'
        WHEN 42 THEN 'dfMONITOR_DATA_TYPE_MONITOR_NETWORK_RECV'
        WHEN 43 THEN 'dfMONITOR_DATA_TYPE_MONITOR_NETWORK_SEND'
        WHEN 44 THEN 'dfMONITOR_DATA_TYPE_MONITOR_AVAILABLE_MEMORY'
	END As Type,
    avg,
    min,
    max
FROM logdb.monitorlog_202412;
